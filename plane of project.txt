# Use official Go image
FROM golang:1.21-alpine

# Set working directory
WORKDIR /app

# Install templ
RUN go install github.com/a-h/templ/cmd/templ@latest

# Copy go mod files first (for caching)
COPY go.mod go.sum ./
RUN go mod download

# Copy all source code
COPY . .

# Generate templ files
RUN templ generate

# Build the app
RUN go build -o gootmplhtmx .

# Expose port (change to your Fiber port)
EXPOSE 3000

# Run the app
CMD ["./gootmplhtmx"]
